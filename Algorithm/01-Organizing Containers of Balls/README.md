    int n = container.size();
    int[] rowSums = new int[n];
    int[] colSums = new int[n];

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            int val = container.get(i).get(j);
            rowSums[i] += val;
            colSums[j] += val;
